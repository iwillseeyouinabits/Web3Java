# Use a lightweight base image\nFROM nginx:alpine\n\n# Create a directory to store the archive\nWORKDIR /usr/share/nginx/html\n\n# Copy the compressed archive into the container\nCOPY archive.zip .\n\n# Install unzip utility\nRUN apk --no-cache add unzip\n\n# Extract the contents of the archive\nRUN unzip archive.zip && rm archive.zip\n\n# Expose the default Nginx port\nEXPOSE 80\n\n# Start Nginx\nCMD ["nginx", "-g", "daemon off;"]
